Introduction to Javascript

JavaScript is a high-level, interpreted language that adds interactivity, logic, and dynamic behavior to web pages and today runs across browsers, servers, mobile apps, and even desktop software..

Where is JavaScript Used?

JavaScript is used in web development to create interactive and dynamic content on websites. It runs in the browser, enabling features like form validation, animations, and user interactions. Additionally, JavaScript can be used on the server side with Node.js, in mobile applications (React Native), desktop applications (Electron), and even in game development.

1. Check Prime

A prime number is only divisible by itself and 1. So, i will run a while loop and increase by 1. (look at the code example. If you dont get it. this is not your cake. do learn javaScript basics and come back.)

Prime.js 


function isPrime(n){
   var divisor = 2;

    while (n > divisor){
        if(n % divisor == 0){
            return false;
        }
        else
            divisor++;
        }

    return true;
}

> isPrime(137);
= true
> isPrime(237);
= false

2. Prime Factors

Run a while loop. start dividing by two and if not divisible increase divider until u r done.

Prime Factors.js 


function primeFactors(n){
  var factors = [], 
      divisor = 2;
  
  while(n>2){
    if(n % divisor == 0){
       factors.push(divisor); 
       n= n/ divisor;
    }
    else{
      divisor++;
    }     
  }
  return factors;
}

> primeFactors(69);
  = [3, 23]

function removeDuplicate(arr){
  var exists ={},
      outArr = [], 
      elm;

  for(var i =0; i < arr.length; i++){
    elm = arr[i];
    if(!exists[elm]){

6. Merge two sorted array

mergeSortedArray.js 


function mergeSortedArray(a, b){
  var merged = [], 
      aElm = a[0],
      bElm = b[0],
      i = 1,
      j = 1;
  
  if(a.length ==0)
    return b;
  if(b.length ==0)
    return a;
  /* 
  if aElm or bElm exists we will insert to merged array
  (will go inside while loop)
   to insert: aElm exists and bElm doesn't exists
             or both exists and aElm < bElm
    this is the critical part of the example            
  */
  while(aElm || bElm){
   if((aElm && !bElm) || aElm < bElm){
     merged.push(aElm);
     aElm = a[i++];
   }   
   else {
     merged.push(bElm);
     bElm = b[j++];
   }
  }
  return merged;
}

> mergeSortedArray([2,5,6,9], [1,2,3,29]);
 = [1, 2, 2, 3, 5, 6, 9, 29]

// Solution 3: Using two pointer swap
function reverse(str) {
  str = str.split('');
  var len = str.length,
      halfIndex = Math.floor(len / 2) - 1,
      revStr;
  for (var i = 0; i <= halfIndex; i++) {
    revStr = str[len - i - 1];
    str[len - i - 1] = str[i];
    str[i] = revStr;
  }
  return str.join('');
}

// Solution 4: Using recursion
function reverse (str) {
    if (str === "") {
        return "";
    } else {
        return reverse(str.substr(1)) + str.charAt(0);
    }
}
  
// Solution 5: Using built-in functions
function reverse(str){
  if(!str || str.length < 2) return str;  
  return str.split('').reverse().join('');
}

// Solution 6: Using ES6 spread operator
const reverse = str => [...str].reverse().join('');

// Solution 7: Using ES6 reduce function
const reverse = str => str.split('').reduce((rev, char) => char + rev, '');


// Solution 8: Using ES6 recursion with destructuring
const reverse = ([first, ...rest]) =>
  first === undefined ? '' : reverse(rest) + first;

14. missing number

missingNumber.js 



function missingNumber(arr){
  var n = arr.length+1, 
  sum = 0,
  expectedSum = n* (n+1)/2;
  
  for(var i = 0, len = arr.length; i < len; i++){
    sum += arr[i];
  }
  
  return expectedSum - sum;
}

17. Counting Zeros

countZero.js 

function countZero(n){
  var count = 0;
  while(n>0){
   count += Math.floor(n/10);
   n = n/10;
  }
  return count;
}

function subStringFinder(str, subStr){
  var idx = 0,
      i = 0,
      j = 0,
      len = str.length,
      subLen = subStr.length;

   for(; i< len; i++){
   
      if(str[i] == subStr[j])
         j++;
      else
         j = 0;
      
      //check starting point or a match   
      if(j == 0)
        idx = i;
      else if (j == subLen)
        return idx;
  }

  return -1;
}

19. Permutations

permutations.js 

function permutations(str){
var arr = str.split(''),
    len = arr.length, 
    perms = [],
    rest,
    picked,
    restPerms,
    next;

    if (len == 0)
        return [str];

    for (var i=0; i< len; i++)
    {
        rest = Object.create(arr);
        picked = rest.splice(i, 1);

        restPerms = permutations(rest.join(''));

       for (var j=0, jLen = restPerms.length; j< jLen; j++)
       {
           next = picked.concat(restPerms[j]);
           perms.push(next.join(''));
       }
    }
   return perms;
}
